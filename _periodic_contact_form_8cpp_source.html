35 void PeriodicContactForm::update_projection() const
36 {
37 const int dim = collision_mesh_.dim();
38 const auto &boundary_vertices = collision_mesh_.rest_positions();
39
40 std::vector<Eigen::Triplet<double>> entries;
41 for (int i = 0; i < collision_mesh_.num_vertices(); i++)
42 {
43 const int i_full = collision_mesh_.to_full_vertex_id(i);
44 for (int d = 0; d < dim; d++)
45 entries.emplace_back(tiled_to_single_(i_full) * dim + d, i_full * dim + d, 1);
46
47 for (int p = 0; p < dim; p++)
48 for (int q = 0; q < dim; q++)
49 entries.emplace_back(n_single_dof_ * dim + p * dim + q, i_full * dim + p, boundary_vertices(i, q));
50 }
51
52 proj.resize(n_single_dof_ * dim + dim * dim, tiled_to_single_.size() * dim);
53 proj.setZero();
54 proj.setFromTriplets(entries.begin(), entries.end());
55 }
