22 T UnconstrainedOgdenElasticity::elastic_energy_T(
23 const RowVectorNd &p,
24 const int el_id,
25 const DefGradMatrix<T> &def_grad) const
26 {
27 constexpr double t = 0; // TODO if we want to allow material that varys over time
28
29 Eigen::Matrix<T, Eigen::Dynamic, 1, 0, 3, 1> eigs;
30
31 if (size() == 2)
32 {
33 // No need to symmetrize F to compute eigen values analytically
34 autogen::eigs_2d<T>(def_grad, eigs);
35 }
36 else
37 {
38 assert(size() == 3);
39 // Symmetrize F to compute eigen values analytically
40 autogen::eigs_3d<T>(def_grad.transpose() * def_grad, eigs);
41 eigs = sqrt(eigs.array());
42 }
43
44 const T J = utils::determinant(def_grad);
45 const T Jdenom = pow(J, -1. / size());
46
47 for (long i = 0; i < eigs.size(); ++i)
48 eigs(i) = eigs(i) * Jdenom;
49
50 auto val = T(0);
51 for (long N = 0; N < alphas_.size(); ++N)
52 {
53 auto tmp = T(-size());
54 const double alpha = alphas_[N](p, t, el_id);
55 const double mu = mus_[N](p, t, el_id);
56
57 for (long i = 0; i < eigs.size(); ++i)
58 tmp += pow(eigs(i), alpha);
59
60 val += 2 * mu / (alpha * alpha) * tmp;
61 }
62
63 for (long N = 0; N < Ds_.size(); ++N)
64 {
65 const double D = Ds_[N](p, t, el_id);
66
67 val += 1. / D * pow(J - T(1), 2 * (N + 1));
68 }
69
70 return val;
71 }

112 template <typename T>
113 T IncompressibleOgdenElasticity::elastic_energy_T(
114 const RowVectorNd &p,
115 const int el_id,
116 const DefGradMatrix<T> &def_grad) const
117 {
118 constexpr double t = 0; // TODO if we want to allow material that varys over time
119
120 Eigen::Matrix<T, Eigen::Dynamic, 1, 0, 3, 1> eigs;
121
122 const T J = polyfem::utils::determinant(def_grad);
123 const T log_J = log(J);
124
125 const auto F_tilde = def_grad / pow(J, 1.0 / size());
126 const auto C_tilde = F_tilde * F_tilde.transpose();
127
128 if (size() == 2)
129 autogen::eigs_2d<T>(C_tilde, eigs);
130 else if (size() == 3)
131 autogen::eigs_3d<T>(C_tilde, eigs);
132 else
133 assert(false);
134 eigs = sqrt(eigs.array());
135
136 T val = T(0);
137 for (long i = 0; i < num_terms(); ++i)
138 {
139 const double c = coefficients_[i](p, t, el_id);
140 const double m = expoenents_[i](p, t, el_id);
141
142 auto tmp = T(-size());
143 for (long j = 0; j < eigs.size(); ++j)
144 tmp += pow(eigs(j), m);
145
146 val += c / (m * m) * tmp;
147 }
148 val += 0.5 * bulk_modulus_(p, t, el_id) * log_J * log_J;
149
150 return val;
151 }
