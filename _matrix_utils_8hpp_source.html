32 template <typename T>
33 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> inverse(const Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> &mat)
34 {
35 assert(mat.rows() == mat.cols());
36
37 if (mat.rows() == 1)
38 {
39 Eigen::Matrix<T, 1, 1> inv;
40 inv(0, 0) = T(1.) / mat(0);
41
42 return inv;
43 }
44 else if (mat.rows() == 2)
45 {
46 Eigen::Matrix<T, 2, 2> inv;
47 T det = determinant(mat);
48 inv(0, 0) = mat(1, 1) / det;
49 inv(0, 1) = -mat(0, 1) / det;
50 inv(1, 0) = -mat(1, 0) / det;
51 inv(1, 1) = mat(0, 0) / det;
52
53 return inv;
54 }
55 else if (mat.rows() == 3)
56 {
57 Eigen::Matrix<T, 3, 3> inv;
58 T det = determinant(mat);
59 inv(0, 0) = (-mat(1, 2) * mat(2, 1) + mat(1, 1) * mat(2, 2)) / det;
60 inv(0, 1) = (mat(0, 2) * mat(2, 1) - mat(0, 1) * mat(2, 2)) / det;
61 inv(0, 2) = (-mat(0, 2) * mat(1, 1) + mat(0, 1) * mat(1, 2)) / det;
62 inv(1, 0) = (mat(1, 2) * mat(2, 0) - mat(1, 0) * mat(2, 2)) / det;
63 inv(1, 1) = (-mat(0, 2) * mat(2, 0) + mat(0, 0) * mat(2, 2)) / det;
64 inv(1, 2) = (mat(0, 2) * mat(1, 0) - mat(0, 0) * mat(1, 2)) / det;
65 inv(2, 0) = (-mat(1, 1) * mat(2, 0) + mat(1, 0) * mat(2, 1)) / det;
66 inv(2, 1) = (mat(0, 1) * mat(2, 0) - mat(0, 0) * mat(2, 1)) / det;
67 inv(2, 2) = (-mat(0, 1) * mat(1, 0) + mat(0, 0) * mat(1, 1)) / det;
68
69 return inv;
70 }
71
72 assert(false);
73 Eigen::Matrix<T, 1, 1> inv;
74 inv(0, 0) = T(0);
75 return inv;
