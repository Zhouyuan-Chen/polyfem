314 const int prev_dim = vertices.cols();
315 vertices.conservativeResize(vertices.rows(), dim);
316 if (prev_dim < dim)
317 vertices.rightCols(dim - prev_dim).setZero();

319 // --------------------------------------------------------------------
320
321 {
322 const VectorNd mesh_dimensions = (vertices.colwise().maxCoeff() - vertices.colwise().minCoeff()).cwiseAbs();
323 MatrixNd A;
324 VectorNd b;
325 construct_affine_transformation(j_mesh["transformation"], mesh_dimensions, A, b);

327 vertices.rowwise() += b.transpose();

332 if (extract == "volume")

336 {
337 // points -> vertices (drop edges and faces)
338 codim_edges.resize(0, 0);
339 faces.resize(0, 0);
340 codim_vertices.resize(vertices.rows());
341 for (int i = 0; i < codim_vertices.size(); ++i)

345 {
346 // edges -> edges (drop faces)

363 // No need to do anything for (extract == "surface" && dim == 3) since we used read_surface_mesh
364 else if (extract == "volume")
365 {
366 // volume -> undefined
367 log_and_throw_error("Volumetric elements not supported for collision obstacles!");
368 }
369
370 if (j_mesh["n_refs"].get<int>() != 0)
371 {
372 log_and_throw_error("Option \"n_refs\" in obstacles not implement yet!");
373 if (j_mesh["advanced"]["refinement_location"].get<double>() != 0.5)
374 log_and_throw_error("Option \"refinement_location\" in obstacles not implement yet!");

571 void construct_affine_transformation(
572 const json &transform,
573 const VectorNd &mesh_dimensions,
574 MatrixNd &A,
575 VectorNd &b)
576 {
577 const int dim = mesh_dimensions.size();
578
579 // -----
580 // Scale
581 // -----
582
583 RowVectorNd scale;
584 if (transform["dimensions"].is_array()) // default is nullptr
585 {
586 VectorNd modified_dimensions =
587 (mesh_dimensions.array() == 0).select(1, mesh_dimensions);
588
589 scale = transform["dimensions"];
590 const int scale_size = scale.size();
591 scale.conservativeResize(dim);
592 if (scale_size < dim)
593 scale.tail(dim - scale_size).setZero();
594
595 scale.array() /= modified_dimensions.array();
596 }
597 else if (transform["scale"].is_number())
598 {
599 scale.setConstant(dim, transform["scale"].get<double>());
600 }
601 else
602 {
603 assert(transform["scale"].is_array());
604 scale = transform["scale"];
605 const int scale_size = scale.size();
606 scale.conservativeResize(dim);
607 if (scale_size < dim)
608 scale.tail(dim - scale_size).setZero();
609
610 if (scale_size == 0)
611 scale.setOnes();
612 }
613
614 A = scale.asDiagonal();
615
616 // ------
617 // Rotate
618 // ------
619
620 // Rotate around the models origin NOT the bodies center of mass.
621 // We could expose this choice as a "rotate_around" field.
622 MatrixNd R = MatrixNd::Identity(dim, dim);
623 if (!transform["rotation"].is_null())
624 {
625 if (dim == 2)
626 {
627 if (transform["rotation"].is_number())
628 R = Eigen::Rotation2Dd(deg2rad(transform["rotation"].get<double>()))
629 .toRotationMatrix();
630 else if (!transform["rotation"].is_array() || !transform["rotation"].empty())
631 log_and_throw_error("Invalid 2D rotation; 2D rotations can only be a angle in degrees.");
632 }
633 else if (dim == 3)
634 {
635 R = to_rotation_matrix(transform["rotation"], transform["rotation_mode"]);
636 }
637 }
638
639 A = R * A; // Scale first, then rotate
640
641 // ---------
642 // Translate
643 // ---------
644
645 b = transform["translation"];
646 const int translation_size = b.size();
647 b.conservativeResize(dim);
648 if (translation_size < dim)
649 b.tail(dim - translation_size).setZero();
650 }
