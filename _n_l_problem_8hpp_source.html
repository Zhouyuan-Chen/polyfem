47 static void full_to_reduced_aux(const State &state, const int full_size, const int reduced_size, const FullMat &full, ReducedMat &reduced)
48 {
49 using namespace polyfem;
50
51 // Reduced is already at the full size
52 if (full_size == reduced_size || full.size() == reduced_size)
53 {
54 reduced = full;
55 return;
56 }
57
58 assert(full.size() == full_size);
59 assert(full.cols() == 1);
60 reduced.resize(reduced_size, 1);
61
62 long j = 0;
63 size_t k = 0;
64 for (int i = 0; i < full.size(); ++i)

66 if (k < state.boundary_nodes.size() && state.boundary_nodes[k] == i)
67 {
68 ++k;
69 continue;
70 }
71
72 reduced(j++) = full(i);
73 }
74 assert(j == reduced.size());
