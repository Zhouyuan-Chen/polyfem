104 void AssemblerUtils::assemble_problem(const std::string &assembler,
105 const bool is_volume,
106 const int n_basis,
107 const std::vector<ElementBases> &bases,
108 const std::vector<ElementBases> &gbases,
109 const AssemblyValsCache &cache,
110 StiffnessMatrix &stiffness) const
111 {
112 if (assembler == "Helmholtz")
113 helmholtz_.assemble(is_volume, n_basis, bases, gbases, cache, stiffness);
114 else if (assembler == "Laplacian")
115 laplacian_.assemble(is_volume, n_basis, bases, gbases, cache, stiffness);
116 else if (assembler == "Bilaplacian")
117 bilaplacian_main_.assemble(is_volume, n_basis, bases, gbases, cache, stiffness);
118
119 else if (assembler == "LinearElasticity")
120 linear_elasticity_.assemble(is_volume, n_basis, bases, gbases, cache, stiffness);
121 else if (assembler == "HookeLinearElasticity")
122 hooke_linear_elasticity_.assemble(is_volume, n_basis, bases, gbases, cache, stiffness);
123 else if (assembler == "Stokes" || assembler == "NavierStokes" || assembler == "OperatorSplitting")
124 stokes_velocity_.assemble(is_volume, n_basis, bases, gbases, cache, stiffness);
125 else if (assembler == "IncompressibleLinearElasticity")
126 incompressible_lin_elast_displacement_.assemble(is_volume, n_basis, bases, gbases, cache, stiffness);
127
128 else if (assembler == "SaintVenant")
129 return;
130 else if (assembler == "NeoHookean")
131 return;
132 else if (assembler == "MooneyRivlin")
133 return;
134 else if (assembler == "MultiModels")
135 return;
136 // else if (assembler == "NavierStokes")
137 // return;
138 else if (assembler == "Ogden")
139 return;
140 else
141 {
142 logger().warn("{} not found, fallback to default", assembler);
143 assert(false);
144 laplacian_.assemble(is_volume, n_basis, bases, gbases, cache, stiffness);
145 }

312 const int el_id,
313 const ElementBases &bs,
314 const ElementBases &gbs,
315 const Eigen::MatrixXd &local_pts,

541 void AssemblerUtils::set_materials(const std::vector<int> &body_ids, const json &body_params)
542 {
543 if (!body_params.is_array())
544 {
545 this->add_multimaterial(0, body_params);
546 return;
547 }
548
549 std::map<int, json> materials;
550 for (int i = 0; i < body_params.size(); ++i)
551 {
552 json mat = body_params[i];
553 json id = mat["id"];
554 if (id.is_array())
555 {
556 for (int j = 0; j < id.size(); ++j)
557 materials[id[j]] = mat;
558 }
559 else
560 {
561 const int mid = id;
562 materials[mid] = mat;
563 }
564 }
565
566 std::set<int> missing;
567
568 std::map<int, int> body_element_count;
569 std::vector<int> eid_to_eid_in_body(body_ids.size());
570 for (int e = 0; e < body_ids.size(); ++e)
571 {
572 const int bid = body_ids[e];
573 body_element_count.try_emplace(bid, 0);
574 eid_to_eid_in_body[e] = body_element_count[bid]++;
575 }
576
577 for (int e = 0; e < body_ids.size(); ++e)
578 {
579 const int bid = body_ids[e];
580 const auto it = materials.find(bid);
581 if (it == materials.end())
582 {
583 missing.insert(bid);
584 continue;
585 }
586
587 const json &tmp = it->second;
588 this->add_multimaterial(e, tmp);
589 }
590
591 for (int bid : missing)
592 {
593 logger().warn("Missing material parameters for body {}", bid);
594 }
595 }

632 void AssemblerUtils::merge_mixed_matrices(
633 const int n_bases, const int n_pressure_bases, const int problem_dim, const bool add_average,
634 const StiffnessMatrix &velocity_stiffness, const StiffnessMatrix &mixed_stiffness, const StiffnessMatrix &pressure_stiffness,
635 StiffnessMatrix &stiffness)
636 {
637 assert(velocity_stiffness.rows() == velocity_stiffness.cols());
638 assert(velocity_stiffness.rows() == n_bases * problem_dim);
639
640 assert(mixed_stiffness.size() == 0 || mixed_stiffness.rows() == n_bases * problem_dim);
641 assert(mixed_stiffness.size() == 0 || mixed_stiffness.cols() == n_pressure_bases);
642
643 assert(pressure_stiffness.size() == 0 || pressure_stiffness.rows() == n_pressure_bases);
644 assert(pressure_stiffness.size() == 0 || pressure_stiffness.cols() == n_pressure_bases);
645
646 const int avg_offset = add_average ? 1 : 0;
647
648 std::vector<Eigen::Triplet<double>> blocks;
649 blocks.reserve(velocity_stiffness.nonZeros() + 2 * mixed_stiffness.nonZeros() + pressure_stiffness.nonZeros() + 2 * avg_offset * velocity_stiffness.rows());
650
651 for (int k = 0; k < velocity_stiffness.outerSize(); ++k)
652 {
653 for (StiffnessMatrix::InnerIterator it(velocity_stiffness, k); it; ++it)
654 {
655 blocks.emplace_back(it.row(), it.col(), it.value());
656 }
657 }
658
659 for (int k = 0; k < mixed_stiffness.outerSize(); ++k)
660 {
661 for (StiffnessMatrix::InnerIterator it(mixed_stiffness, k); it; ++it)
662 {
663 blocks.emplace_back(it.row(), n_bases * problem_dim + it.col(), it.value());
664 blocks.emplace_back(it.col() + n_bases * problem_dim, it.row(), it.value());
665 }
666 }
667
668 for (int k = 0; k < pressure_stiffness.outerSize(); ++k)
669 {
670 for (StiffnessMatrix::InnerIterator it(pressure_stiffness, k); it; ++it)
671 {
672 blocks.emplace_back(n_bases * problem_dim + it.row(), n_bases * problem_dim + it.col(), it.value());
673 }
674 }
675
676 if (add_average)
677 {
678 const double val = 1.0 / n_pressure_bases;
679 for (int i = 0; i < n_pressure_bases; ++i)
680 {
681 blocks.emplace_back(n_bases * problem_dim + i, n_bases * problem_dim + n_pressure_bases, val);
682 blocks.emplace_back(n_bases * problem_dim + n_pressure_bases, n_bases * problem_dim + i, val);
683 }
684 }
685
686 stiffness.resize(n_bases * problem_dim + n_pressure_bases + avg_offset, n_bases * problem_dim + n_pressure_bases + avg_offset);
687 stiffness.setFromTriplets(blocks.begin(), blocks.end());
688 stiffness.makeCompressed();
689
690 // static int c = 0;
691 // Eigen::saveMarket(stiffness, "stiffness.txt");
692 // Eigen::saveMarket(velocity_stiffness, "velocity_stiffness.txt");
693 // Eigen::saveMarket(mixed_stiffness, "mixed_stiffness.txt");
694 // Eigen::saveMarket(pressure_stiffness, "pressure_stiffness.txt");
695 }

697 int AssemblerUtils::quadrature_order(const std::string &assembler, const int basis_degree, const BasisType &b_type, const int dim)
698 {
699 if (assembler == "Mass")
700 {
701 if (b_type == BasisType::SIMPLEX_LAGRANGE || b_type == BasisType::CUBE_LAGRANGE)
702 return std::max(basis_degree * 2, 1);
703 else
704 return basis_degree * 2 + 1;
705 }
706 else if (assembler == "NavierStokes")
707 {
708 if (b_type == BasisType::SIMPLEX_LAGRANGE)
709 return std::max((basis_degree - 1) + basis_degree, 1);
710 else if (b_type == BasisType::CUBE_LAGRANGE)
711 return std::max(basis_degree * 2, 1);
712 else
713 return basis_degree * 2 + 1;
714 }
715 else
716 {
717 if (b_type == BasisType::SIMPLEX_LAGRANGE)
718 return std::max((basis_degree - 1) * 2, 1);
719 else if (b_type == BasisType::CUBE_LAGRANGE)
720 return std::max(basis_degree * 2, 1);
721 else
722 return (basis_degree - 1) * 2 + 1;
723 }
724 }

726 std::map<std::string, AssemblerUtils::ParamFunc> AssemblerUtils::parameters(const std::string &assembler) const
727 {
728 std::map<std::string, ParamFunc> res;
729
730 // "Laplacian" "Bilaplacian" "Damping" "MultiModels"
731
732 if (assembler == "Helmholtz")
733 {
734 const double k = helmholtz_.local_assembler().k();
735 res["k"] = [k](const RowVectorNd &, const RowVectorNd &, double, int) { return k; };
736 }
737 else if (assembler == "LinearElasticity" || assembler == "IncompressibleLinearElasticity" || assembler == "NeoHookean")
738 {
739 const auto &params = linear_elasticity_.local_assembler().lame_params();
740 const int size = linear_elasticity_.local_assembler().size();
741
742 res["lambda"] = [&params](const RowVectorNd &uv, const RowVectorNd &p, double t, int e) {
743 double lambda, mu;
744
745 params.lambda_mu(uv, p, e, lambda, mu);
746 return lambda;
747 };
748
749 res["mu"] = [&params](const RowVectorNd &uv, const RowVectorNd &p, double t, int e) {
750 double lambda, mu;
751
752 params.lambda_mu(uv, p, e, lambda, mu);
753 return mu;
754 };
755
756 res["E"] = [&params, size](const RowVectorNd &uv, const RowVectorNd &p, double t, int e) {
757 double lambda, mu;
758 params.lambda_mu(uv, p, e, lambda, mu);
759
760 if (size == 3)
761 return mu * (3.0 * lambda + 2.0 * mu) / (lambda + mu);
762 else
763 return 2 * mu * (2.0 * lambda + 2.0 * mu) / (lambda + 2.0 * mu);
764 };
765
766 res["nu"] = [&params, size](const RowVectorNd &uv, const RowVectorNd &p, double t, int e) {
767 double lambda, mu;
768
769 params.lambda_mu(uv, p, e, lambda, mu);
770
771 if (size == 3)
772 return lambda / (2.0 * (lambda + mu));
773 else
774 return lambda / (lambda + 2.0 * mu);
775 };
776 }
777 else if (assembler == "HookeLinearElasticity" || assembler == "SaintVenant")
778 {
779 const auto &elast_tensor = hooke_linear_elasticity_.local_assembler().elasticity_tensor();
780 const int size = hooke_linear_elasticity_.local_assembler().size() == 2 ? 3 : 6;
781
782 for (int i = 0; i < size; ++i)
783 {
784 for (int j = i; j < size; ++j)
785 {
786 res[fmt::format("C_{}{}", i, j)] = [&elast_tensor, i, j](const RowVectorNd &, const RowVectorNd &, double, int) {
787 return elast_tensor(i, j);
788 };
789 }
790 }
791 }
792 else if (assembler == "MooneyRivlin")
793 {
794 const auto &c1 = mooney_rivlin_elasticity_.local_assembler().formulation().c1();
795 const auto &c2 = mooney_rivlin_elasticity_.local_assembler().formulation().c2();
796 const auto &k = mooney_rivlin_elasticity_.local_assembler().formulation().k();
797
798 res["c1"] = [&c1](const RowVectorNd &, const RowVectorNd &p, double t, int e) {
799 return c1(p, t, e);
800 };
801
802 res["c2"] = [&c2](const RowVectorNd &, const RowVectorNd &p, double t, int e) {
803 return c2(p, t, e);
804 };
805
806 res["k"] = [&k](const RowVectorNd &, const RowVectorNd &p, double t, int e) {
807 return k(p, t, e);
808 };
809 }
810 else if (assembler == "Ogden")
811 {
812 const Eigen::VectorXd alphas = ogden_elasticity_.local_assembler().formulation().alphas();
813 const Eigen::VectorXd mus = ogden_elasticity_.local_assembler().formulation().mus();
814 const Eigen::VectorXd Ds = ogden_elasticity_.local_assembler().formulation().Ds();
815
816 for (int i = 0; i < alphas.size(); ++i)
817 res[fmt::format("alpha_{}", i)] = [&alphas, i](const RowVectorNd &, const RowVectorNd &, double, int) { return alphas[i]; };
818
819 for (int i = 0; i < mus.size(); ++i)
820 res[fmt::format("mu_{}", i)] = [&mus, i](const RowVectorNd &, const RowVectorNd &, double, int) { return mus[i]; };
821
822 for (int i = 0; i < Ds.size(); ++i)
823 res[fmt::format("D_{}", i)] = [&Ds, i](const RowVectorNd &, const RowVectorNd &, double, int) { return Ds[i]; };
824 }
825 else if (assembler == "Stokes" || assembler == "NavierStokes" || assembler == "OperatorSplitting")
826 {
827 const double nu = stokes_velocity_.local_assembler().viscosity();
828 res["viscosity"] = [nu](const RowVectorNd &, const RowVectorNd &, double, int) { return nu; };
829 }
830
831 return res;
832 }
