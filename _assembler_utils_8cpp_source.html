169 else if (assembler == "Laplacian")
170 laplacian_->assemble(is_volume, n_basis, bases, gbases, cache, stiffness);
171 else if (assembler == "Bilaplacian")
172 bilaplacian_main_->assemble(is_volume, n_basis, bases, gbases, cache, stiffness);

174 else if (assembler == "LinearElasticity")
175 linear_elasticity_->assemble(is_volume, n_basis, bases, gbases, cache, stiffness);
176 else if (assembler == "HookeLinearElasticity")
177 hooke_linear_elasticity_->assemble(is_volume, n_basis, bases, gbases, cache, stiffness);
178 else if (assembler == "Stokes" || assembler == "NavierStokes" || assembler == "OperatorSplitting")
179 stokes_velocity_->assemble(is_volume, n_basis, bases, gbases, cache, stiffness);
180 else if (assembler == "IncompressibleLinearElasticity")
181 incompressible_lin_elast_displacement_->assemble(is_volume, n_basis, bases, gbases, cache, stiffness);
182
183 else if (assembler == "SaintVenant")
184 return;
185 else if (assembler == "NeoHookean")
186 return;
187 else if (assembler == "MooneyRivlin")
188 return;
189 else if (assembler == "MultiModels")
190 return;
191 // else if (assembler == "NavierStokes")
192 // return;
193 else if (assembler == "UnconstrainedOgden")
194 return;
195 else if (assembler == "IncompressibleOgden")
196 return;
197 else
198 {
199 logger().warn("{} not found, fallback to default", assembler);
200 assert(false);
201 laplacian_->assemble(is_volume, n_basis, bases, gbases, cache, stiffness);
202 }

669 {

681 if (id.is_array())

688 const int mid = id;

695 std::map<int, int> body_element_count;

703
704 for (int e = 0; e < body_ids.size(); ++e)

710 missing.insert(bid);

712 }
713
714 const json &tmp = it->second;
715 this->add_multimaterial(e, tmp);
716 }
717
718 for (int bid : missing)

759 void AssemblerUtils::merge_mixed_matrices(
760 const int n_bases, const int n_pressure_bases, const int problem_dim, const bool add_average,
761 const StiffnessMatrix &velocity_stiffness, const StiffnessMatrix &mixed_stiffness, const StiffnessMatrix &pressure_stiffness,
762 StiffnessMatrix &stiffness)
763 {
764 assert(velocity_stiffness.rows() == velocity_stiffness.cols());
765 assert(velocity_stiffness.rows() == n_bases * problem_dim);
766
767 assert(mixed_stiffness.size() == 0 || mixed_stiffness.rows() == n_bases * problem_dim);
768 assert(mixed_stiffness.size() == 0 || mixed_stiffness.cols() == n_pressure_bases);
769
770 assert(pressure_stiffness.size() == 0 || pressure_stiffness.rows() == n_pressure_bases);
771 assert(pressure_stiffness.size() == 0 || pressure_stiffness.cols() == n_pressure_bases);
772
773 const int avg_offset = add_average ? 1 : 0;
774
775 std::vector<Eigen::Triplet<double>> blocks;
776 blocks.reserve(velocity_stiffness.nonZeros() + 2 * mixed_stiffness.nonZeros() + pressure_stiffness.nonZeros() + 2 * avg_offset * velocity_stiffness.rows());
777
778 for (int k = 0; k < velocity_stiffness.outerSize(); ++k)
779 {
780 for (StiffnessMatrix::InnerIterator it(velocity_stiffness, k); it; ++it)
781 {
782 blocks.emplace_back(it.row(), it.col(), it.value());
783 }
784 }
785
786 for (int k = 0; k < mixed_stiffness.outerSize(); ++k)
787 {
788 for (StiffnessMatrix::InnerIterator it(mixed_stiffness, k); it; ++it)
789 {
790 blocks.emplace_back(it.row(), n_bases * problem_dim + it.col(), it.value());
791 blocks.emplace_back(it.col() + n_bases * problem_dim, it.row(), it.value());
792 }
793 }
794
795 for (int k = 0; k < pressure_stiffness.outerSize(); ++k)
796 {
797 for (StiffnessMatrix::InnerIterator it(pressure_stiffness, k); it; ++it)
798 {
799 blocks.emplace_back(n_bases * problem_dim + it.row(), n_bases * problem_dim + it.col(), it.value());
800 }
801 }
802
803 if (add_average)
804 {
805 const double val = 1.0 / n_pressure_bases;
806 for (int i = 0; i < n_pressure_bases; ++i)
807 {
808 blocks.emplace_back(n_bases * problem_dim + i, n_bases * problem_dim + n_pressure_bases, val);
809 blocks.emplace_back(n_bases * problem_dim + n_pressure_bases, n_bases * problem_dim + i, val);
810 }
811 }
812
813 stiffness.resize(n_bases * problem_dim + n_pressure_bases + avg_offset, n_bases * problem_dim + n_pressure_bases + avg_offset);
814 stiffness.setFromTriplets(blocks.begin(), blocks.end());
815 stiffness.makeCompressed();

871
872 params.lambda_mu(uv, p, e, lambda, mu);

878

884 double lambda, mu;

886
887 if (size == 3)
888 return mu * (3.0 * lambda + 2.0 * mu) / (lambda + mu);
889 else
890 return 2 * mu * (2.0 * lambda + 2.0 * mu) / (lambda + 2.0 * mu);
891 };
892
893 res["nu"] = [&params, size](const RowVectorNd &uv, const RowVectorNd &p, double t, int e) {
894 double lambda, mu;
895
896 params.lambda_mu(uv, p, e, lambda, mu);
897
898 if (size == 3)
899 return lambda / (2.0 * (lambda + mu));
900 else
901 return lambda / (lambda + 2.0 * mu);
902 };
