219 template <typename T>
220 T SaintVenantElasticity::compute_energy_aux(const NonLinearAssemblerData &data) const
221 {
222 typedef Eigen::Matrix<T, Eigen::Dynamic, 1> AutoDiffVect;
223 typedef Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> AutoDiffGradMat;
224
225 AutoDiffVect local_disp;
226 get_local_disp(data, size(), local_disp);
227
228 AutoDiffGradMat disp_grad(size(), size());
229
230 T energy = T(0.0);
231
232 const int n_pts = data.da.size();
233 for (long p = 0; p < n_pts; ++p)
234 {
235 compute_disp_grad_at_quad(data, local_disp, p, size(), disp_grad);
236
237 AutoDiffGradMat strain = strain_from_disp_grad(disp_grad);
238 AutoDiffGradMat stress_tensor(size(), size());
239
240 if (size() == 2)
241 {
242 std::array<T, 3> eps;
243 eps[0] = strain(0, 0);
244 eps[1] = strain(1, 1);
245 eps[2] = 2 * strain(0, 1);
246
247 stress_tensor << stress(eps, 0), stress(eps, 2),
248 stress(eps, 2), stress(eps, 1);
249 }
250 else
251 {
252 std::array<T, 6> eps;
253 eps[0] = strain(0, 0);
254 eps[1] = strain(1, 1);
255 eps[2] = strain(2, 2);
256 eps[3] = 2 * strain(1, 2);
257 eps[4] = 2 * strain(0, 2);
258 eps[5] = 2 * strain(0, 1);
259
260 stress_tensor << stress(eps, 0), stress(eps, 5), stress(eps, 4),
261 stress(eps, 5), stress(eps, 1), stress(eps, 3),
262 stress(eps, 4), stress(eps, 3), stress(eps, 2);
263 }
264
265 energy += (stress_tensor * strain).trace() * data.da(p);
266 }
267
268 return energy * 0.5;
