127 template <typename Derived>
128 void GenericElastic<Derived>::compute_stress_grad_multiply_mat(
129 const int el_id,
130 const Eigen::MatrixXd &local_pts,
131 const Eigen::MatrixXd &global_pts,
132 const Eigen::MatrixXd &grad_u_i,
133 const Eigen::MatrixXd &mat,
134 Eigen::MatrixXd &stress,
135 Eigen::MatrixXd &result) const
136 {
137 typedef DScalar2<double, Eigen::Matrix<double, Eigen::Dynamic, 1, 0, 9, 1>, Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic, 0, 9, 9>> Diff;
138
139 DiffScalarBase::setVariableCount(size() * size());
140 Eigen::Matrix<Diff, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> def_grad(size(), size());
141
142 Eigen::MatrixXd F = grad_u_i;
143 for (int d = 0; d < size(); ++d)
144 F(d, d) += 1.;
145
146 assert(local_pts.rows() == 1);
147 for (int i = 0; i < size(); ++i)
148 for (int j = 0; j < size(); ++j)
149 def_grad(i, j) = Diff(i + j * size(), F(i, j));
150
151 auto energy = derived().elastic_energy(global_pts, el_id, def_grad);
152
153 // Grad is ∂W(F)/∂F_ij
154 Eigen::MatrixXd grad = energy.getGradient().reshaped(size(), size());
155 // Hessian is ∂W(F)/(∂F_ij*∂F_kl)
156 Eigen::MatrixXd hess = energy.getHessian();
157
158 // Stress is S_ij = ∂W(F)/∂F_ij
159 stress = grad;
160 // Compute ∂S_ij/∂F_kl * M_kl, same as M_ij * ∂S_ij/∂F_kl since the hessian is symmetric
161 result = (hess * mat.reshaped(size() * size(), 1)).reshaped(size(), size());
162 }
