131 if (x.size() < 100)
132 {
133 polyfem::logger().debug("x: {}", x.transpose());

135 polyfem::logger().debug("direc: {}", direction.transpose());

140 reset_history(x.size());
141 increase_descent_strategy();
142 polyfem::logger().log(

147 }

153 this->descent_strategy == 2 ? spdlog::level::warn : spdlog::level::debug,
154 "L-BFGS direction is not a descent direction (Δx⋅g={}); reverting to {}",

156 return compute_update_direction(objFunc, x, grad, direction);
157 }
158
159 m_prev_x = x;
160 m_prev_grad = grad;
161
