14 std::shared_ptr<Selection> Selection::build(
15 const json &selection,
16 const Selection::BBox &mesh_bbox,
17 const std::string &root_path)
18 {
19 if (!selection.contains("id"))
20 {
21 logger().error("Selection does not contain an id field: {}", selection.dump());
22 throw std::runtime_error("Selection id not found");
23 }
24
25 std::shared_ptr<Selection> res = nullptr;
26 if (selection.contains("box"))
27 res = std::make_shared<BoxSelection>(selection, mesh_bbox);
28 else if (selection.contains("center"))
29 res = std::make_shared<SphereSelection>(selection, mesh_bbox);
30 else if (selection.contains("axis"))
31 res = std::make_shared<AxisPlaneSelection>(selection, mesh_bbox);
32 else if (selection.contains("normal"))
33 res = std::make_shared<PlaneSelection>(selection, mesh_bbox);
34 else if (selection["id"].is_string()) // assume ID is a file path
35 res = std::make_shared<FileSelection>(
36 resolve_path(selection["id"], root_path), selection.value("id_offset", 0));
37 else if (selection["id"].is_number_integer()) // assume ID is uniform
38 res = std::make_shared<UniformSelection>(selection["id"]);
39 else
40 log_and_throw_error(fmt::format("Selection not recognized: {}", selection.dump()));
41
42 return res;
43 }

151 AxisPlaneSelection::AxisPlaneSelection(
152 const json &selection,
153 const Selection::BBox &mesh_bbox)
154 : Selection(selection["id"].get<int>())
155 {
156 position_ = selection["position"];
157
158 if (selection["axis"].is_string())
159 {
160 std::string axis = selection["axis"];
161 int sign = axis[0] == '-' ? -1 : 1;
162 int dim = std::tolower(axis.back()) - 'x' + 1;
163 assert(dim >= 1 && dim <= 3);
164 axis_ = sign * dim;
165 }
166 else
167 {
168 assert(selection["axis"].is_number_integer());
169 axis_ = selection["axis"];
170 assert(std::abs(axis_) >= 1 && std::abs(axis_) <= 3);
171 }
172
173 if (selection.value("relative", false))
174 {
175 int dim = std::abs(axis_) - 1;
176 position_ = (mesh_bbox[1][dim] - mesh_bbox[0][dim]) * position_ + mesh_bbox[0][dim];
177 }
178 }
