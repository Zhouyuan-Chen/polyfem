28 IncompressibleLinearElasticityDispacement::assemble(const LinearAssemblerData &data) const
29 {
30 // 2mu (epsi : epsj)
31 assert(size_ == 2 || size_ == 3);
32
33 Eigen::Matrix<double, Eigen::Dynamic, 1, 0, 9, 1> res(size() * size());
34 res.setZero();
35
36 const Eigen::MatrixXd &gradi = data.vals.basis_values[data.i].grad_t_m;
37 const Eigen::MatrixXd &gradj = data.vals.basis_values[data.j].grad_t_m;
38
39 Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> epsi(size(), size());
40 Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> epsj(size(), size());
41
42 for (long p = 0; p < gradi.rows(); ++p)
43 {
44
45 double lambda, mu;
46 params_.lambda_mu(data.vals.quadrature.points.row(p), data.vals.val.row(p), data.vals.element_id, lambda, mu);
47
48 for (int di = 0; di < size(); ++di)
49 {
50 epsi.setZero();
51 epsi.row(di) = gradi.row(p);
52 epsi = ((epsi + epsi.transpose()) / 2).eval();
53 for (int dj = 0; dj < size(); ++dj)
54 {
55 epsj.setZero();
56 epsj.row(dj) = gradj.row(p);
57 epsj = ((epsj + epsj.transpose()) / 2.0).eval();
58
59 res(dj * size() + di) += 2 * mu * (epsi.array() * epsj.array()).sum() * data.da(p);
60 }
61 }
62 }
63
64 return res;
65 }
