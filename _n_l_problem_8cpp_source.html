7u^{t+1} = u(t+\Delta t)\approx u(t) + \Delta t \dot u + \frac{\Delta t^2} 2 \ddot u \newline
8= u(t) + \Delta t \dot u + \frac{\Delta t^2}{2} \psi\newline
9M u^{t+1}_h \approx M u^t_h + \Delta t M v^t_h + \frac{\Delta t^2} {2} A u^{t+1}_h \newline

169 void NLProblem::full_to_reduced_aux(const std::vector<int> &boundary_nodes, const int full_size, const int reduced_size, const FullMat &full, ReducedMat &reduced)
170 {
171 using namespace polyfem;
172
173 // Reduced is already at the full size
174 if (full_size == reduced_size || full.size() == reduced_size)
175 {
176 reduced = full;
177 return;
178 }
179
180 assert(full.size() == full_size);
181 assert(full.cols() == 1);
182 reduced.resize(reduced_size, 1);
183
184 assert(std::is_sorted(boundary_nodes.begin(), boundary_nodes.end()));
185
186 long j = 0;
187 size_t k = 0;
188 for (int i = 0; i < full.size(); ++i)
189 {
190 if (k < boundary_nodes.size() && boundary_nodes[k] == i)
191 {
192 ++k;
193 continue;
194 }
195
196 reduced(j++) = full(i);
197 }
198 assert(j == reduced.size());
199 }
