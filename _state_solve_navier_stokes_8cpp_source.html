145
146 /* apply boundary condition */
147 solve_data.rhs_assembler->set_bc(
148 local_boundary, boundary_nodes, n_b_samples, local_neumann_boundary, sol, Eigen::MatrixXd(), time);

150 /* export to vtu */
151 save_timestep(time, t, 0, dt, sol, pressure);
