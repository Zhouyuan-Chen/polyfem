124int forward_simulation(const CLI::App &command_line,
125 const std::string &hdf5_file,
126 const std::string output_dir,
127 const size_t max_threads,
128 const bool is_strict,
129 const bool fallback_solver,
130 const spdlog::level::level_enum &log_level,
131 json &in_args)
132{
133 std::vector<std::string> names;
134 std::vector<Eigen::MatrixXi> cells;
135 std::vector<Eigen::MatrixXd> vertices;
136
137 if (in_args.empty() && hdf5_file.empty())
138 {
139 logger().error("No input file specified!");
140 return command_line.exit(CLI::RequiredError("--json or --hdf5"));
141 }
142
143 if (in_args.empty() && !hdf5_file.empty())
144 {
145 using MatrixXl = Eigen::Matrix<int64_t, Eigen::Dynamic, Eigen::Dynamic>;
146
147 h5pp::File file(hdf5_file, h5pp::FileAccess::READONLY);
148 std::string json_string = file.readDataset<std::string>("json");
149
150 in_args = json::parse(json_string);
151 in_args["root_path"] = hdf5_file;
152
153 names = file.findGroups("", "/meshes");
154 cells.resize(names.size());
155 vertices.resize(names.size());
156
157 for (size_t i = 0; i < names.size(); ++i)
158 {
159 const std::string &name = names[i];
160 cells[i] = file.readDataset<MatrixXl>("/meshes/" + name + "/c").cast<int>();
161 vertices[i] = file.readDataset<Eigen::MatrixXd>("/meshes/" + name + "/v");
162 }
163 }
164
165 json tmp = json::object();
166 if (has_arg(command_line, "log_level"))
167 tmp["/output/log/level"_json_pointer] = int(log_level);
168 if (has_arg(command_line, "max_threads"))
169 tmp["/solver/max_threads"_json_pointer] = max_threads;
170 if (has_arg(command_line, "output_dir"))
171 tmp["/output/directory"_json_pointer] = std::filesystem::absolute(output_dir);
172 if (has_arg(command_line, "enable_overwrite_solver"))
173 tmp["/solver/linear/enable_overwrite_solver"_json_pointer] = fallback_solver;
174 assert(tmp.is_object());
175 in_args.merge_patch(tmp);
176
177 State state;
178 state.init(in_args, is_strict);
179 state.load_mesh(/*non_conforming=*/false, names, cells, vertices);
180
181 // Mesh was not loaded successfully; load_mesh() logged the error.
182 if (state.mesh == nullptr)
183 {
184 // Cannot proceed without a mesh.
185 return EXIT_FAILURE;
186 }
187
188 state.stats.compute_mesh_stats(*state.mesh);
189
190 state.build_basis();
191
192 state.assemble_rhs();
193 state.assemble_mass_mat();
194
195 Eigen::MatrixXd sol;
196 Eigen::MatrixXd pressure;
197
198 state.solve_problem(sol, pressure);
199
200 state.compute_errors(sol);
201
202 logger().info("total time: {}s", state.timings.total_time());
203
204 state.save_json(sol);
205 state.export_data(sol, pressure);
206
207 return EXIT_SUCCESS;
208}
