65 template <typename ProblemType>
66 bool SparseNewtonDescentSolver<ProblemType>::compute_update_direction(
67 ProblemType &objFunc,
68 const TVector &x,
69 const TVector &grad,
70 TVector &direction)
71 {
72 if (this->descent_strategy == 2)
73 {
74 direction = -grad;
75 return true;
76 }
77
78 polyfem::StiffnessMatrix hessian;
79
80 assemble_hessian(objFunc, x, hessian);
81
82 if (!solve_linear_system(hessian, grad, direction))
83 // solve_linear_system will increase descent_strategy if needed
84 return compute_update_direction(objFunc, x, grad, direction);
85
86 if (!check_direction(hessian, grad, direction))
87 // check_direction will increase descent_strategy if needed
88 return compute_update_direction(objFunc, x, grad, direction);
89
90 json info;
91 linear_solver->getInfo(info);
92 internal_solver_info.push_back(info);
93
94 reg_weight /= reg_weight_dec;
95 if (reg_weight < reg_weight_min)
96 reg_weight = 0;
97
98 return true;
99 }

103 template <typename ProblemType>
104 void SparseNewtonDescentSolver<ProblemType>::assemble_hessian(
105 ProblemType &objFunc, const TVector &x, polyfem::StiffnessMatrix &hessian)
106 {
107 POLYFEM_SCOPED_TIMER("assembly time", this->assembly_time);
108
109 if (this->descent_strategy == 1)
110 objFunc.set_project_to_psd(true);
111 else if (this->descent_strategy == 0)
112 objFunc.set_project_to_psd(false);
113 else
114 assert(false);
115
116 objFunc.hessian(x, hessian);
117
118 if (reg_weight > 0)
119 {
120 hessian += reg_weight * polyfem::utils::sparse_identity(hessian.rows(), hessian.cols());
121 }
122 }

126 template <typename ProblemType>
127 bool SparseNewtonDescentSolver<ProblemType>::solve_linear_system(
128 const polyfem::StiffnessMatrix &hessian, const TVector &grad, TVector &direction)
129 {
130 POLYFEM_SCOPED_TIMER("linear solve", this->inverting_time);
131 // TODO: get the correct size
132 linear_solver->analyzePattern(hessian, hessian.rows());
133
134 try
135 {
136 linear_solver->factorize(hessian);
137 }
138 catch (const std::runtime_error &err)
139 {
140 increase_descent_strategy();
141
142 // warn if using gradient descent
143 polyfem::logger().log(
144 log_level(), "Unable to factorize Hessian: \"{}\"; reverting to {}",
145 err.what(), this->descent_strategy_name());
146
147 // polyfem::write_sparse_matrix_csv("problematic_hessian.csv", hessian);
148 return false;
149 }
150
151 linear_solver->solve(-grad, direction); // H Δx = -g
152
153 return true;
154 }

158 template <typename ProblemType>
159 bool SparseNewtonDescentSolver<ProblemType>::check_direction(
160 const polyfem::StiffnessMatrix &hessian, const TVector &grad, const TVector &direction)
161 {
162 // gradient descent, check descent direction
163 const double residual = (hessian * direction + grad).norm(); // H Δx + g = 0
164 if (std::isnan(residual) || residual > std::max(1e-8 * grad.norm(), 1e-5) * characteristic_length)
165 {
166 increase_descent_strategy();
167
168 polyfem::logger().log(
169 log_level(),
170 "[{}] large (or nan) linear solve residual {} (||∇f||={}); reverting to {}",
171 name(), residual, grad.norm(), this->descent_strategy_name());
172
173 return false;
174 }
175 else
176 {
177 polyfem::logger().trace("linear solve residual {}", residual);
178 }
179
180 // do this check here because we need to repeat the solve without resetting reg_weight
181 if (grad.dot(direction) >= 0)
182 {
183 increase_descent_strategy();
184 polyfem::logger().log(
185 log_level(), "[{}] direction is not a descent direction (Δx⋅g={}≥0); reverting to {}",
186 name(), direction.dot(grad), descent_strategy_name());
187 return false;
188 }
189
190 return true;
191 }
