218 void SolveData::update_barrier_stiffness(const Eigen::VectorXd &x)
219 {
220 if (contact_form == nullptr || !contact_form->use_adaptive_barrier_stiffness())
221 return;
222
223 Eigen::VectorXd grad_energy = Eigen::VectorXd::Zero(x.size());
224 const std::array<std::shared_ptr<Form>, 3> energy_forms{
225 {elastic_form, inertia_form, body_form}};
226 for (const std::shared_ptr<Form> &form : energy_forms)
227 {
228 if (form == nullptr || !form->enabled())
229 continue;
230
231 Eigen::VectorXd grad_form;
232 form->first_derivative(x, grad_form);
233 grad_energy += grad_form;
234 }
235
236 contact_form->update_barrier_stiffness(x, grad_energy);
237 }

239 void SolveData::update_dt()
240 {
241 if (time_integrator == nullptr) // if is not time dependent
242 return;
243
244 const std::array<std::shared_ptr<Form>, 5> energy_forms{
245 {elastic_form, body_form, damping_form, contact_form, friction_form}};
246 for (const std::shared_ptr<Form> &form : energy_forms)
247 {
248 if (form == nullptr)
249 continue;
250 form->set_weight(time_integrator->acceleration_scaling());
251 }
