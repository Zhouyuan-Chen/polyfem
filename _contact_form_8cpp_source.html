108 double ContactForm::max_step_size(const Eigen::VectorXd &x0, const Eigen::VectorXd &x1) const
109 {
110 // Extract surface only
111 const Eigen::MatrixXd V0 = compute_displaced_surface(x0);
112 const Eigen::MatrixXd V1 = compute_displaced_surface(x1);
113
114 double max_step;
115 if (use_cached_candidates_ && broad_phase_method_ != ipc::BroadPhaseMethod::SWEEP_AND_TINIEST_QUEUE_GPU)
116 max_step = ipc::compute_collision_free_stepsize(
117 candidates_, state_.collision_mesh, V0, V1, ccd_tolerance_, ccd_max_iterations_);
118 else
119 max_step = ipc::compute_collision_free_stepsize(
120 state_.collision_mesh, V0, V1, broad_phase_method_, ccd_tolerance_, ccd_max_iterations_);
121
122#ifndef NDEBUG
123 // This will check for static intersections as a failsafe. Not needed if we use our conservative CCD.
124 Eigen::MatrixXd V_toi = (V1 - V0) * max_step + V0;
125
126 while (ipc::has_intersections(state_.collision_mesh, V_toi))
127 {
128 logger().error("taking max_step results in intersections (max_step={:g})", max_step);
129 max_step /= 2.0;
130
131 const double Linf = (V_toi - V0).lpNorm<Eigen::Infinity>();
132 if (max_step <= 0 || Linf == 0)
133 log_and_throw_error(fmt::format("Unable to find an intersection free step size (max_step={:g} L∞={:g})", max_step, Linf));
134
135 V_toi = (V1 - V0) * max_step + V0;
136 }
137#endif
138
139 return max_step;
140 }

194 bool ContactForm::is_step_collision_free(const Eigen::VectorXd &x0, const Eigen::VectorXd &x1) const
195 {
196 const auto displaced0 = compute_displaced_surface(x0);
197 const auto displaced1 = compute_displaced_surface(x1);
198
199 // Skip CCD if the displacement is zero.
200 if ((displaced1 - displaced0).lpNorm<Eigen::Infinity>() == 0.0)
201 {
202 // Assumes initially intersection-free
203 return true;
204 }
205
206 bool is_valid;
207 if (use_cached_candidates_)
208 is_valid = ipc::is_step_collision_free(
209 candidates_, state_.collision_mesh,
210 displaced0,
211 displaced1,
212 ccd_tolerance_, ccd_max_iterations_);
213 else
214 is_valid = ipc::is_step_collision_free(
215 state_.collision_mesh,
216 displaced0,
217 displaced1,
218 broad_phase_method_, ccd_tolerance_, ccd_max_iterations_);
219
220 return is_valid;
