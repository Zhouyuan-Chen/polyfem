72 {
73 m_prev_x.resize(ndof);
74 m_prev_grad.resize(ndof);
75
76 hess.setIdentity(ndof, ndof);
77
78 // Use gradient descent for first iteration

82 virtual bool compute_update_direction(
83 ProblemType &objFunc,
84 const TVector &x,
85 const TVector &grad,
86 TVector &direction) override
87 {
88 if (this->descent_strategy == 2)
89 {
90 direction = -grad;
91 }
92 else
93 {
94 direction = hess.ldlt().solve(-grad);
95
96 TVector y = grad - m_prev_grad;
97 TVector s = x - m_prev_x;
98
99 double y_s = y.dot(s);
100 TVector Bs = hess * s;
101 double sBs = s.transpose() * Bs;
102
103 hess += (y * y.transpose()) / y_s - (Bs * Bs.transpose()) / sBs;
104 }
105
106 m_prev_x = x;
107 m_prev_grad = grad;
108
109 if (std::isnan(direction.squaredNorm()))
110 {
111 reset_history(x.size());
112 increase_descent_strategy();
113 polyfem::logger().log(
114 this->descent_strategy == 2 ? spdlog::level::warn : spdlog::level::debug,
115 "nan in direction {} (||∇f||={}); reverting to {}",
116 direction.dot(grad), this->descent_strategy_name());
117 return compute_update_direction(objFunc, x, grad, direction);
118 }
119 else if (grad.squaredNorm() != 0 && direction.dot(grad) >= 0)
120 {
121 reset_history(x.size());
122 increase_descent_strategy();
123 polyfem::logger().log(
124 this->descent_strategy == 2 ? spdlog::level::warn : spdlog::level::debug,
125 "L-BFGS direction is not a descent direction (Δx⋅g={}≥0); reverting to {}",
126 direction.dot(grad), this->descent_strategy_name());
127 return compute_update_direction(objFunc, x, grad, direction);
128 }
129
130 return true;
131 }
132 };
