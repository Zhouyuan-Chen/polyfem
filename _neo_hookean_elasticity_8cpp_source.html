258 T NeoHookeanElasticity::compute_energy_aux(const NonLinearAssemblerData &data) const
259 {
260 typedef Eigen::Matrix<T, Eigen::Dynamic, 1> AutoDiffVect;
261 typedef Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> AutoDiffGradMat;
262
263 AutoDiffVect local_disp;
264 get_local_disp(data, size(), local_disp);
265
266 AutoDiffGradMat def_grad(size(), size());
267
268 T energy = T(0.0);
269
270 const int n_pts = data.da.size();
271 for (long p = 0; p < n_pts; ++p)
272 {
273 compute_disp_grad_at_quad(data, local_disp, p, size(), def_grad);
274
275 // Id + grad d
276 for (int d = 0; d < size(); ++d)
277 def_grad(d, d) += T(1);
278
279 double lambda, mu;
280 params_.lambda_mu(data.vals.quadrature.points.row(p), data.vals.val.row(p), data.vals.element_id, lambda, mu);
281
282 const T log_det_j = log(polyfem::utils::determinant(def_grad));
283 const T val = mu / 2 * ((def_grad.transpose() * def_grad).trace() - size() - 2 * log_det_j) + lambda / 2 * log_det_j * log_det_j;

285 energy += val * data.da(p);

291 Eigen::Matrix<double, dim, dim> hat(const Eigen::Matrix<double, dim, 1> &x)
292 {
293
294 Eigen::Matrix<double, dim, dim> prod;
295 prod.setZero();
296
297 prod(0, 1) = -x(2);
298 prod(0, 2) = x(1);
299 prod(1, 0) = x(2);
300 prod(1, 2) = -x(0);
301 prod(2, 0) = -x(1);
302 prod(2, 1) = x(0);
303
304 return prod;

322 void NeoHookeanElasticity::compute_energy_aux_gradient_fast(const NonLinearAssemblerData &data, Eigen::Matrix<double, Eigen::Dynamic, 1> &G_flattened) const
323 {
324 assert(data.x.cols() == 1);
325
326 const int n_pts = data.da.size();
327
328 Eigen::Matrix<double, n_basis, dim> local_disp(data.vals.basis_values.size(), size());
329 local_disp.setZero();
330 for (size_t i = 0; i < data.vals.basis_values.size(); ++i)
331 {
332 const auto &bs = data.vals.basis_values[i];
333 for (size_t ii = 0; ii < bs.global.size(); ++ii)
334 {
335 for (int d = 0; d < size(); ++d)
336 {
337 local_disp(i, d) += bs.global[ii].val * data.x(bs.global[ii].index * size() + d);
338 }
339 }
340 }
341
342 Eigen::Matrix<double, dim, dim> def_grad(size(), size());
343
344 Eigen::Matrix<double, n_basis, dim> G(data.vals.basis_values.size(), size());
345 G.setZero();
346
347 for (long p = 0; p < n_pts; ++p)
348 {
349 Eigen::Matrix<double, n_basis, dim> grad(data.vals.basis_values.size(), size());
350
351 for (size_t i = 0; i < data.vals.basis_values.size(); ++i)
352 {
353 grad.row(i) = data.vals.basis_values[i].grad.row(p);
354 }
355
356 Eigen::Matrix<double, dim, dim> jac_it = data.vals.jac_it[p];
357
358 // Id + grad d
359 def_grad = local_disp.transpose() * grad * jac_it + Eigen::Matrix<double, dim, dim>::Identity(size(), size());
360
361 const double J = def_grad.determinant();
362 const double log_det_j = log(J);
363
364 Eigen::Matrix<double, dim, dim> delJ_delF(size(), size());
365 delJ_delF.setZero();
366
367 if (dim == 2)
368 {
369
370 delJ_delF(0, 0) = def_grad(1, 1);
371 delJ_delF(0, 1) = -def_grad(1, 0);
372 delJ_delF(1, 0) = -def_grad(0, 1);
373 delJ_delF(1, 1) = def_grad(0, 0);
374 }
375
376 else if (dim == 3)
377 {
378
379 Eigen::Matrix<double, dim, 1> u(def_grad.rows());
380 Eigen::Matrix<double, dim, 1> v(def_grad.rows());
381 Eigen::Matrix<double, dim, 1> w(def_grad.rows());
382
383 u = def_grad.col(0);
384 v = def_grad.col(1);
385 w = def_grad.col(2);
386
387 delJ_delF.col(0) = cross<dim>(v, w);
388 delJ_delF.col(1) = cross<dim>(w, u);
389 delJ_delF.col(2) = cross<dim>(u, v);
390 }
391
392 double lambda, mu;
393 params_.lambda_mu(data.vals.quadrature.points.row(p), data.vals.val.row(p), data.vals.element_id, lambda, mu);
394
395 Eigen::Matrix<double, n_basis, dim> delF_delU = grad * jac_it;
396
397 Eigen::Matrix<double, dim, dim> gradient_temp = mu * def_grad - mu * (1 / J) * delJ_delF + lambda * log_det_j * (1 / J) * delJ_delF;
398 Eigen::Matrix<double, n_basis, dim> gradient = delF_delU * gradient_temp.transpose();
399
400 double val = mu / 2 * ((def_grad.transpose() * def_grad).trace() - size() - 2 * log_det_j) + lambda / 2 * log_det_j * log_det_j;
401
402 G.noalias() += gradient * data.da(p);
403 }
404
405 Eigen::Matrix<double, dim, n_basis> G_T = G.transpose();
406
407 constexpr int N = (n_basis == Eigen::Dynamic) ? Eigen::Dynamic : n_basis * dim;
408 Eigen::Matrix<double, N, 1> temp(Eigen::Map<Eigen::Matrix<double, N, 1>>(G_T.data(), G_T.size()));
409 G_flattened = temp;
410 }

478 w = def_grad.col(2);

524 void NeoHookeanElasticity::compute_dstress_dgradu_multiply_mat(const int el_id, const Eigen::MatrixXd &local_pts, const Eigen::MatrixXd &global_pts, const Eigen::MatrixXd &grad_u_i, const Eigen::MatrixXd &mat, Eigen::MatrixXd &stress, Eigen::MatrixXd &result) const
525 {

528
529 Eigen::MatrixXd def_grad = Eigen::MatrixXd::Identity(grad_u_i.rows(), grad_u_i.cols()) + grad_u_i;

531
532 stress = mu * (def_grad - FmT) + lambda * std::log(def_grad.determinant()) * FmT;
