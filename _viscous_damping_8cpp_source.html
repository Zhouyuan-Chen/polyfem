256 }

261 for (long k = 0; k < jac_it.size(); ++k)
262 jac_it(k) = data.vals.jac_it[p](k);
263
264 def_grad = def_grad * jac_it + Eigen::MatrixXd::Identity(size(), size());

266

268 Eigen::MatrixXd dEdt = 0.5 * (dFdt.transpose() * def_grad + def_grad.transpose() * dFdt);
