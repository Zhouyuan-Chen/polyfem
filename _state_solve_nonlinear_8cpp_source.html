46 void SolveData::updated_barrier_stiffness(const Eigen::VectorXd &x)
47 {
48 // TODO: missing use_adaptive_barrier_stiffness_ if (use_adaptive_barrier_stiffness_ && is_time_dependent_)

73 contact_form->update_barrier_stiffness(x, grad_energy);

311 void State::solve_tensor_nonlinear(const int t)
312 {
313
314 assert(solve_data.nl_problem != nullptr);
315 NLProblem &nl_problem = *(solve_data.nl_problem);
316
317 assert(sol.size() == rhs.size());
318
319 if (nl_problem.uses_lagging())
320 {
321 POLYFEM_SCOPED_TIMER("Initializing lagging");
322 nl_problem.init_lagging(sol);
323 logger().info("Lagging iteration {:d}:", 1);
324 }
325
326 // ---------------------------------------------------------------------
327
328 // Save the subsolve sequence for debugging
329 int subsolve_count = 0;
330 save_subsolve(subsolve_count, t);
331
332 // ---------------------------------------------------------------------
333
334 std::shared_ptr<cppoptlib::NonlinearSolver<NLProblem>> nl_solver = make_nl_solver<NLProblem>();
335
336 ALSolver al_solver(
337 nl_solver, solve_data.al_form,
338 args["solver"]["augmented_lagrangian"]["initial_weight"],
339 args["solver"]["augmented_lagrangian"]["max_weight"],
340 [&](const Eigen::VectorXd &x) {
341 this->solve_data.updated_barrier_stiffness(sol);
342 });
343
344 al_solver.post_subsolve = [&](const double al_weight) {
345 json info;
346 nl_solver->get_info(info);
347 stats.solver_info.push_back(
348 {{"type", al_weight > 0 ? "al" : "rc"},
349 {"t", t}, // TODO: null if static?
350 {"info", info}});
351 if (al_weight > 0)
352 stats.solver_info.back()["weight"] = al_weight;
353 save_subsolve(++subsolve_count, t);
354 };
355
356 al_solver.solve(nl_problem, sol, args["solver"]["augmented_lagrangian"]["force"]);
357
358 // ---------------------------------------------------------------------
359
360 // TODO: Make this more general
361 const double lagging_tol = args["solver"]["contact"].value("friction_convergence_tol", 1e-2);
362
363 // Lagging loop (start at 1 because we already did an iteration above)
364 bool lagging_converged = !nl_problem.uses_lagging();
365 for (int lag_i = 1; !lagging_converged; lag_i++)
366 {
367 Eigen::VectorXd tmp_sol = nl_problem.full_to_reduced(sol);
368
369 // Update the lagging before checking for convergence
370 nl_problem.update_lagging(tmp_sol, lag_i);
371
372 // Check if lagging converged
373 Eigen::VectorXd grad;
374 nl_problem.gradient(tmp_sol, grad);
375 logger().debug("Lagging convergence grad_norm={:g} tol={:g}", grad.norm(), lagging_tol);
376 if (grad.norm() <= lagging_tol)
377 {
378 logger().info(
379 "Lagging converged in {:d} iteration(s) (grad_norm={:g} tol={:g})",
380 lag_i, grad.norm(), lagging_tol);
381 lagging_converged = true;
382 break;
383 }
384
385 // Check for convergence first before checking if we can continue
386 if (lag_i >= nl_problem.max_lagging_iterations())
387 {
388 logger().warn(
389 "Lagging failed to converge with {:d} iteration(s) (grad_norm={:g} tol={:g})",
390 lag_i, grad.norm(), lagging_tol);
391 lagging_converged = false;
392 break;
393 }
394
395 // Solve the problem with the updated lagging
396 logger().info("Lagging iteration {:d}:", lag_i + 1);
397 nl_problem.init(sol);
398 solve_data.updated_barrier_stiffness(sol);
399 nl_solver->minimize(nl_problem, tmp_sol);
400 sol = nl_problem.reduced_to_full(tmp_sol);
401
402 // Save the subsolve sequence for debugging and info
403 json info;
404 nl_solver->get_info(info);
405 stats.solver_info.push_back(
406 {{"type", "rc"},
407 {"t", t}, // TODO: null if static?
408 {"lag_i", lag_i},
409 {"info", info}});
410 save_subsolve(++subsolve_count, t);
411 }
