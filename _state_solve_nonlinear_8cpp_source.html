121 void State::solve_tensor_nonlinear(const int t)
122 {
123 Eigen::VectorXd tmp_sol;
124
125 assert(solve_data.nl_problem != nullptr);
126 NLProblem &nl_problem = *(solve_data.nl_problem);
127 assert(solve_data.alnl_problem != nullptr);
128 ALNLProblem &alnl_problem = *(solve_data.alnl_problem);
129
130 assert(sol.size() == rhs.size());
131
132 double al_weight = args["solver"]["augmented_lagrangian"]["initial_weight"];
133 const double max_al_weight = args["solver"]["augmented_lagrangian"]["max_weight"];
134

136 assert(sol.size() == rhs.size());
137 assert(tmp_sol.size() <= rhs.size());
138
139 {
140 POLYFEM_SCOPED_TIMER("Initializing lagging");
141 nl_problem.init_lagging(sol);

143 }
144
145 const int friction_iterations = args["solver"]["contact"]["friction_iterations"];
146 assert(friction_iterations >= 0);
147 if (friction_iterations > 0)
148 logger().debug("Lagging iteration 1");
149
150 // Disable damping for the final lagged iteration
151 if (friction_iterations <= 1)
152 {
153 nl_problem.lagged_damping_weight() = 0;
154 alnl_problem.lagged_damping_weight() = 0;
155 }
156
157 // Save the subsolve sequence for debugging
158 int subsolve_count = 0;
159 save_subsolve(subsolve_count, t);
160
162
163 nl_problem.line_search_begin(sol, tmp_sol);
164 bool force_al = args["solver"]["augmented_lagrangian"]["force"];
165 while (force_al || !std::isfinite(nl_problem.value(tmp_sol)) || !nl_problem.is_step_valid(sol, tmp_sol)
166 || !nl_problem.is_step_collision_free(sol, tmp_sol))
167 {
168 force_al = false;
169 nl_problem.line_search_end();
170 alnl_problem.set_weight(al_weight);
171 logger().debug("Solving AL Problem with weight {}", al_weight);
172
173 std::shared_ptr<cppoptlib::NonlinearSolver<ALNLProblem>> alnl_solver = make_nl_solver<ALNLProblem>();
174 alnl_solver->setLineSearch(args["solver"]["nonlinear"]["line_search"]["method"]);
175 alnl_problem.init(sol);
176 tmp_sol = sol;
177 alnl_solver->minimize(alnl_problem, tmp_sol);
178 json alnl_solver_info;
179 alnl_solver->getInfo(alnl_solver_info);
180
181 solver_info.push_back(
182 {{"type", "al"},
183 {"t", t}, // TODO: null if static?
184 {"weight", al_weight},
185 {"info", alnl_solver_info}});
186
187 sol = tmp_sol;
188 nl_problem.full_to_reduced(sol, tmp_sol);
189 nl_problem.line_search_begin(sol, tmp_sol);
190
191 al_weight *= 2;
192
193 if (al_weight >= max_al_weight)

197 logger().error(msg);

200 }
201
202 save_subsolve(++subsolve_count, t);
203 }
204 nl_problem.line_search_end();
205
207

209 nl_solver->setLineSearch(args["solver"]["nonlinear"]["line_search"]["method"]);
210 nl_problem.init(sol);
211 nl_solver->minimize(nl_problem, tmp_sol);
212 json nl_solver_info;
213 nl_solver->getInfo(nl_solver_info);
214 solver_info.push_back(
215 {{"type", "rc"},

217 {"info", nl_solver_info}});
218 nl_problem.reduced_to_full(tmp_sol, sol);
219
220 save_subsolve(++subsolve_count, t);
221
223
224 // TODO: fix this
225 nl_problem.lagged_damping_weight() = 0;
226
227 // Lagging loop (start at 1 because we already did an iteration above)
228 int lag_i;
229 nl_problem.update_lagging(tmp_sol);
230 bool lagging_converged = nl_problem.lagging_converged(tmp_sol);
231 for (lag_i = 1; !lagging_converged && lag_i < friction_iterations; lag_i++)
232 {
233 logger().debug("Lagging iteration {:d}", lag_i + 1);
234 nl_problem.init(sol);
235 // Disable damping for the final lagged iteration
236 if (lag_i == friction_iterations - 1)
237 nl_problem.lagged_damping_weight() = 0;
238 nl_solver->minimize(nl_problem, tmp_sol);
239
240 nl_solver->getInfo(nl_solver_info);
241 solver_info.push_back(
242 {{"type", "rc"},
243 {"t", t}, // TODO: null if static?
244 {"lag_i", lag_i},
245 {"info", nl_solver_info}});
246
247 nl_problem.reduced_to_full(tmp_sol, sol);
248 nl_problem.update_lagging(tmp_sol);
249 lagging_converged = nl_problem.lagging_converged(tmp_sol);
250
251 save_subsolve(++subsolve_count, t);
252 }
253
255
256 if (friction_iterations > 0)
257 {
258 logger().log(
259 lagging_converged ? spdlog::level::info : spdlog::level::warn,
260 "{} {:d} lagging iteration(s) (err={:g} tol={:g})",
261 lagging_converged ? "Friction lagging converged using" : "Friction lagging maxed out at", lag_i,
262 nl_problem.compute_lagging_error(tmp_sol),
263 args["solver"]["contact"]["friction_convergence_tol"].get<double>());
264 }
