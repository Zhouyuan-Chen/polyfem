95void MMASolverAux::SolveDIP(double *x) {
96
97 for (int j = 0; j < m; j++) {
98 lam[j] = c[j] / 2.0;
99 mu[j] = 1.0;
100 }
101
102 const double tol = epsimin; // 1.0e-9*sqrt(m+n);
103 double epsi = 1.0;
104 double err = 1.0;
105 int loop;
106
107 while (epsi > tol) {
108
109 loop = 0;
110 while (err > 0.9 * epsi && loop < 100) {
111 loop++;
112
113 // Set up Newton system
114 XYZofLAMBDA(x);
115 DualGrad(x);
116 for (int j = 0; j < m; j++) {
117 grad[j] = -1.0 * grad[j] - epsi / lam[j];
118 }
119 DualHess(x);
120
121 // Solve Newton system
122 if (m > 1) {
123 Factorize(hess.data(), m);
124 Solve(hess.data(), grad.data(), m);
125 for (int j = 0; j < m; j++) {
126 s[j] = grad[j];
127 }
128 } else if (m > 0) {
129 s[0] = grad[0] / hess[0];
130 }
131
132 // Get the full search direction
133 for (int i = 0; i < m; i++) {
134 s[m + i] = -mu[i] + epsi / lam[i] - s[i] * mu[i] / lam[i];
135 }
136
137 // Perform linesearch and update lam and mu
138 DualLineSearch();
139
140 XYZofLAMBDA(x);
141
142 // Compute KKT res
143 err = DualResidual(x, epsi);
144 }
145 epsi = epsi * 0.1;
146 }
147}

149void MMASolverAux::SolveDSA(double *x) {
150
151 for (int j = 0; j < m; j++) {
152 lam[j] = 1.0;
153 }
154
155 const double tol = epsimin; // 1.0e-9*sqrt(m+n);
156 double err = 1.0;
157 int loop = 0;
158
159 while (err > tol && loop < 500) {
160 loop++;
161 XYZofLAMBDA(x);
162 DualGrad(x);
163 double theta = 1.0;
164 err = 0.0;
165 for (int j = 0; j < m; j++) {
166 lam[j] = std::max(0.0, lam[j] + theta * grad[j]);
167 err += grad[j] * grad[j];
168 }
169 err = std::sqrt(err);
170 }
171}

173double MMASolverAux::DualResidual(double *x, double epsi) {
174
175 double *res = new double[2 * m];
176
177 for (int j = 0; j < m; j++) {
178 res[j] = -b[j] - a[j] * z - y[j] + mu[j];
179 res[j + m] = mu[j] * lam[j] - epsi;
180 for (int i = 0; i < n; i++) {
181 res[j] += pij[i * m + j] / (upp[i] - x[i]) + qij[i * m + j] / (x[i] - low[i]);
182 }
183 }
184
185 double nrI = 0.0;
186 for (int i = 0; i < 2 * m; i++) {
187 if (nrI < std::abs(res[i])) {
188 nrI = std::abs(res[i]);
189 }
190 }
191
192 delete[] res;
193
194 return nrI;
195}

197void MMASolverAux::DualLineSearch() {
198
199 double theta = 1.005;
200 for (int i = 0; i < m; i++) {
201 if (theta < -1.01 * s[i] / lam[i]) {
202 theta = -1.01 * s[i] / lam[i];
203 }
204 if (theta < -1.01 * s[i + m] / mu[i]) {
205 theta = -1.01 * s[i + m] / mu[i];
206 }
207 }
208 theta = 1.0 / theta;
209
210 for (int i = 0; i < m; i++) {
211 lam[i] = lam[i] + theta * s[i];
212 mu[i] = mu[i] + theta * s[i + m];
213 }

216void MMASolverAux::DualHess(double *x) {
217
218 double *df2 = new double[n];
219 double *PQ = new double[n * m];
220 #ifdef MMA_WITH_OPENMP
221 #pragma omp parallel for
222 #endif
223 for (int i = 0; i < n; i++) {
224 double pjlam = p0[i];
225 double qjlam = q0[i];
226 for (int j = 0; j < m; j++) {
227 pjlam += pij[i * m + j] * lam[j];
228 qjlam += qij[i * m + j] * lam[j];
229 PQ[i * m + j] = pij[i * m + j] / pow(upp[i] - x[i], 2.0) - qij[i * m + j] / pow(x[i] - low[i], 2.0);
230 }
231 df2[i] = -1.0 / (2.0 * pjlam / pow(upp[i] - x[i], 3.0) + 2.0 * qjlam / pow(x[i] - low[i], 3.0));
232 double xp = (sqrt(pjlam) * low[i] + sqrt(qjlam) * upp[i]) / (sqrt(pjlam) + sqrt(qjlam));
233 if (xp < alpha[i]) {
234 df2[i] = 0.0;
235 }
236 if (xp > beta[i]) {
237 df2[i] = 0.0;
238 }
239 }
240
241 // Create the matrix/matrix/matrix product: PQ^T * diag(df2) * PQ
242 double *tmp = new double[n * m];
243 for (int j = 0; j < m; j++) {
244 #ifdef MMA_WITH_OPENMP
245 #pragma omp parallel for
246 #endif
247 for (int i = 0; i < n; i++) {
248 tmp[j * n + i] = 0.0;
249 tmp[j * n + i] += PQ[i * m + j] * df2[i];
250 }
251 }
252
253 for (int i = 0; i < m; i++) {
254 for (int j = 0; j < m; j++) {
255 hess[i * m + j] = 0.0;
256 for (int k = 0; k < n; k++) {
257 hess[i * m + j] += tmp[i * n + k] * PQ[k * m + j];
258 }
259 }
260 }
261
262 double lamai = 0.0;
263 for (int j = 0; j < m; j++) {
264 if (lam[j] < 0.0) {
265 lam[j] = 0.0;
266 }
267 lamai += lam[j] * a[j];
268 if (lam[j] > c[j]) {
269 hess[j * m + j] += -1.0;
270 }
271 hess[j * m + j] += -mu[j] / lam[j];
272 }
273
274 if (lamai > 0.0) {
275 for (int j = 0; j < m; j++) {
276 for (int k = 0; k < m; k++) {
277 hess[j * m + k] += -10.0 * a[j] * a[k];
278 }
279 }
280 }
281
282 // pos def check
283 double HessTrace = 0.0;
284 for (int i = 0; i < m; i++) {
285 HessTrace += hess[i * m + i];
286 }
287 double HessCorr = 1e-4 * HessTrace / m;
288
289 if (-1.0 * HessCorr < 1.0e-7) {
290 HessCorr = -1.0e-7;
291 }
292
293 for (int i = 0; i < m; i++) {
294 hess[i * m + i] += HessCorr;
295 }
296
297 delete[] df2;
298 delete[] PQ;
299 delete[] tmp;
300}

311void MMASolverAux::XYZofLAMBDA(double *x) {
312
313 double lamai = 0.0;
314 for (int i = 0; i < m; i++) {
315 if (lam[i] < 0.0) {
316 lam[i] = 0;
317 }
318 y[i] = std::max(0.0, lam[i] - c[i]); // Note y=(lam-c)/d - however d is fixed at one !!
319 lamai += lam[i] * a[i];
320 }
321 z = std::max(0.0, 10.0 * (lamai - 1.0)); // SINCE a0 = 1.0
322
323 #ifdef MMA_WITH_OPENMP
324 #pragma omp parallel for
325 #endif
326 for (int i = 0; i < n; i++) {
327 double pjlam = p0[i];
328 double qjlam = q0[i];
329 for (int j = 0; j < m; j++) {
330 pjlam += pij[i * m + j] * lam[j];
331 qjlam += qij[i * m + j] * lam[j];
332 }
333 x[i] = (sqrt(pjlam) * low[i] + sqrt(qjlam) * upp[i]) / (sqrt(pjlam) + sqrt(qjlam));
334 if (x[i] < alpha[i]) {
335 x[i] = alpha[i];
336 }
337 if (x[i] > beta[i]) {
338 x[i] = beta[i];
339 }
340 }

343void MMASolverAux::GenSub(const double *xval, const double *dfdx, const double *gx, const double *dgdx, const double *xmin,
344 const double *xmax)
345{
346 // Forward the iterator
347 iter++;
348
349 // Set asymptotes
350 if (iter < 3) {
351 #ifdef MMA_WITH_OPENMP
352 #pragma omp parallel for
353 #endif
354 for (int i = 0; i < n; i++) {
355 low[i] = xval[i] - asyminit * (xmax[i] - xmin[i]);
356 upp[i] = xval[i] + asyminit * (xmax[i] - xmin[i]);
357 }
358 } else {
359 #ifdef MMA_WITH_OPENMP
360 #pragma omp parallel for
361 #endif
362 for (int i = 0; i < n; i++) {
363 double zzz = (xval[i] - xold1[i]) * (xold1[i] - xold2[i]);
364 double gamma;
365 if (zzz < 0.0) {
366 gamma = asymdec;
367 } else if (zzz > 0.0) {
368 gamma = asyminc;
369 } else {
370 gamma = 1.0;
371 }
372 low[i] = xval[i] - gamma * (xold1[i] - low[i]);
373 upp[i] = xval[i] + gamma * (upp[i] - xold1[i]);
374
375 double xmami = std::max(xmamieps, xmax[i] - xmin[i]);
376 // double xmami = xmax[i] - xmin[i];
377 low[i] = std::max(low[i], xval[i] - 100.0 * xmami);
378 low[i] = std::min(low[i], xval[i] - 1.0e-5 * xmami);
379 upp[i] = std::max(upp[i], xval[i] + 1.0e-5 * xmami);
380 upp[i] = std::min(upp[i], xval[i] + 100.0 * xmami);
381
382 double xmi = xmin[i] - 1.0e-6;
383 double xma = xmax[i] + 1.0e-6;
384 if (xval[i] < xmi) {
385 low[i] = xval[i] - (xma - xval[i]) / 0.9;
386 upp[i] = xval[i] + (xma - xval[i]) / 0.9;
387 }
388 if (xval[i] > xma) {
389 low[i] = xval[i] - (xval[i] - xmi) / 0.9;
390 upp[i] = xval[i] + (xval[i] - xmi) / 0.9;
391 }
392 }
393 }
394
395 // Set bounds and the coefficients for the approximation
396 // double raa0 = 0.5*1e-6;
397 #ifdef MMA_WITH_OPENMP
398 #pragma omp parallel for
399 #endif
400 for (int i = 0; i < n; ++i) {
401 // Compute bounds alpha and beta
402 alpha[i] = std::max(xmin[i], low[i] + albefa * (xval[i] - low[i]));
403 alpha[i] = std::max(alpha[i], xval[i] - move * (xmax[i] - xmin[i]));
404 alpha[i] = std::min(alpha[i], xmax[i]);
405 beta[i] = std::min(xmax[i], upp[i] - albefa * (upp[i] - xval[i]));
406 beta[i] = std::min(beta[i], xval[i] + move * (xmax[i] - xmin[i]));
407 beta[i] = std::max(beta[i], xmin[i]);
408
409 // Objective function
410 {
411 double dfdxp = std::max(0.0, dfdx[i]);
412 double dfdxm = std::max(0.0, -1.0 * dfdx[i]);
413 double xmamiinv = 1.0 / std::max(xmamieps, xmax[i] - xmin[i]);
414 double pq = 0.001 * std::abs(dfdx[i]) + raa0 * xmamiinv;
415 p0[i] = std::pow(upp[i] - xval[i], 2.0) * (dfdxp + pq);
416 q0[i] = std::pow(xval[i] - low[i], 2.0) * (dfdxm + pq);
417 }
418
419 // Constraints
420 for (int j = 0; j < m; j++) {
421 double dgdxp = std::max(0.0, dgdx[i * m + j]);
422 double dgdxm = std::max(0.0, -1.0 * dgdx[i * m + j]);
423 double xmamiinv = 1.0 / std::max(xmamieps, xmax[i] - xmin[i]);
424 double pq = 0.001 * std::abs(dgdx[i * m + j]) + raa0 * xmamiinv;
425 pij[i * m + j] = std::pow(upp[i] - xval[i], 2.0) * (dgdxp + pq);
426 qij[i * m + j] = std::pow(xval[i] - low[i], 2.0) * (dgdxm + pq);
427 }
428 }
429
430 // The constant for the constraints
431 for (int j = 0; j < m; j++) {
432 b[j] = -gx[j];
433 for (int i = 0; i < n; i++) {
434 b[j] += pij[i * m + j] / (upp[i] - xval[i]) + qij[i * m + j] / (xval[i] - low[i]);
435 }
436 }
437}

439void MMASolverAux::Factorize(double *K, int n) {
440
441 for (int s = 0; s < n - 1; s++) {
442 for (int i = s + 1; i < n; i++) {
443 K[i * n + s] = K[i * n + s] / K[s * n + s];
444 for (int j = s + 1; j < n; j++) {
445 K[i * n + j] = K[i * n + j] - K[i * n + s] * K[s * n + j];
446 }
447 }
448 }

451void MMASolverAux::Solve(double *K, double *x, int n) {
452
453 for (int i = 1; i < n; i++) {
454 double a = 0.0;
455 for (int j = 0; j < i; j++) {
456 a = a - K[i * n + j] * x[j];
457 }
458 x[i] = x[i] + a;
459 }
460
461 x[n - 1] = x[n - 1] / K[(n - 1) * n + (n - 1)];
462 for (int i = n - 2; i >= 0; i--) {
463 double a = x[i];
464 for (int j = i + 1; j < n; j++) {
465 a = a - K[i * n + j] * x[j];
466 }
467 x[i] = a / K[i * n + i];
468 }
